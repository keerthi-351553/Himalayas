let addUser = (req, res) => {
  let newUser = new user({
    Username : req.body.username,
    Password : req.body.password,
    Loginid : req.body.loginid,
    Role : req.body.role
  });
  newUser.save().then((docs) => {
    logger.debug(docs);
    res.send(docs);
  }, (err) => {
    res.status(400).send(err);
    logger.debug('error occurred while adding');
  });
};
